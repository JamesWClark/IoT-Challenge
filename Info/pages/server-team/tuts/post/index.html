Step 2: Verify the contents of your package.json file:

  "dependencies": {
    "body-parser": "^1.17.1",
    "express": "^4.15.2"
  }
}
Step 3: Require the body-parser software at the top of your code:
var bodyParser = require('body-parser');
Step 4: Use the body parser middleware before you write any routes:
app.use(bodyParser.json());
app.use(bodyParser.urlencoded({ extended: false }));
Step 5: Write a POST route for the home page:
app.post('/', function(req, res) {
    console.log('post / = ' + req.body);
    res.status(200).send('got it');
});
Step 6: This is what server.js should look like now:
var bodyParser = require('body-parser');
var express = require('express');

var app = express();

app.use(bodyParser.json());
app.use(bodyParser.urlencoded({ extended: false }));

app.get('/', function (req, res) {
  res.send('Hello World!');
});

app.post('/', function(req, res) {
    console.log('post / = ' + JSON.stringify(req.body));
    res.status(200).send('got it');
});

app.listen(3000, function () {
  console.log('Example app listening on port 3000!');
});
Step 7: Run your server. If it's already running from a previous exercise, cancel it with Control + C and run it again.
