Step 1: Use npm to install the latest version of socket.io.
npm install socket.io --save
Step 2: Verify the contents of package.json:
{
  "dependencies": {
    "body-parser": "^1.17.1",
    "express": "^4.15.2",
    "socket.io": "^1.7.3"
  }
}
Step 3: In your server.js file, find the line that reads var app = express();.

Step 4: Immediately following it, insert this line which creates an HTTP server from the express app object.
var server = http.createServer(app);
Step 5: Then, attach web sockets to the server object:
var io = socketio(server);
Step 6: Create a POST route called echo.
app.post('/echo', function(req, res) {
    console.log('post /echo = ' + JSON.stringify(req.body));
    io.sockets.emit('echo', req.body);
    res.status(201).send(req.body);
});
Step 7: Test the echo route with ARC.

Step 8: Finally, change that last line to listen on the server instead of the app object.
server.listen(3000, function () {
    console.log('Example app listening on port 3000!');
});
Step 9: Your server.js file should by now be similar to the following:
var bodyParser = require('body-parser');
var express = require('express');
var http = require('http');
var socketio = require('socket.io');

var app = express();
var server = http.createServer(app);
var io = socketio(server);

app.use(bodyParser.json());
app.use(bodyParser.urlencoded({ extended: false }));

app.post('/', function(req, res) {
    console.log('post / = ' + JSON.stringify(req.body));
    res.status(200).send('got it');
});

app.post('/echo', function(req, res) {
    console.log('post /echo = ' + JSON.stringify(req.body));
    io.sockets.emit('echo', req.body);
    res.status(201).send(req.body);
});

server.listen(3000, function () {
    console.log('Example app listening on port 3000!');
});
Step 9: To complete this task, two things need to happen: 1) The sensors team needs to send JSON data through this echo route, and 2) the clients team needs to connect a web socket to receive your broadcoast relay from the sensors team.
